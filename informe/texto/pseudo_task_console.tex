\subsection{Pseudocodigo}

\begin{algorithm}[H]
\caption{TaskConsola(pid, params)}
\label{pseudo:TaskConsola}
\begin{algorithmic}

\STATE n = params[0]
\STATE bmin = params[1]
\STATE bmax = params[2]
\FOR{$i=0$ hasta $n$}
    \STATE uso\_IO(pid, bmin + rand() \% (bmax - bmin +1))
\ENDFOR

\end{algorithmic}
\end{algorithm}
